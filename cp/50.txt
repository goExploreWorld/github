jQuery(document).ready(function() {

    //login
    if (typeof(ajax_login_params) == 'undefined') {
        initAjaxLogin();
    }
    else {
        initAjaxLogin(ajax_login_params);
    }

    //stats
    var urlParser = new LHC.URLParser(window.location.href);
    var queryStringParser = new LHC.QueryStringParser(urlParser.getQueryString());
    var agentId = parseInt(queryStringParser.get('agentId'));
    var cookie = jQuery.cookie(C.COOKIE_USERSTATS);
    if( !cookie ) {
        var stats_url = G.baseUrl + '/stats.php';
        if (!isNaN(agentId) && agentId > 0) {
            stats_url += '?agentId=' + agentId;
        }
        jQuery('body').append('<div style="display:none;"><' + 'iframe src' + '="' + stats_url + '"></iframe></div>');
    }

    (function(){
        //feedback link
        var feedback_position = function(){
            var bottom = 10;
            var jQuery_doc = jQuery(document);
            var jQuery_win = jQuery(window);
            var doc_scrollTop = jQuery_doc.scrollTop();
            var win_height = jQuery_win.height();
            var doc_height = jQuery_doc.height();
            var jQuery_feedback_link = jQuery('#feedback_link');
            if (G.browser.msie && parseInt(G.browser.version, 10) == 6) {
                jQuery_feedback_link.css({position: 'absolute', top: (win_height - jQuery_feedback_link.height() + doc_scrollTop - bottom) + 'px'});
            }
            else {
                jQuery_feedback_link.css({position: 'fixed' , bottom : bottom + 'px'});
            }
        };

        // 二维码
        var two_dimensional_position = function(){
            var bottom = 100;
            var jQuery_doc = jQuery(document);
            var jQuery_win = jQuery(window);
            var doc_scrollTop = jQuery_doc.scrollTop();
            var win_height = jQuery_win.height();
            var doc_height = jQuery_doc.height();
            var two_dimensional = jQuery('#two_dimensional');
            var pageWidth = 990;
            var clientWidth = document.documentElement.clientWidth;
            var adWidth = 100;
            var positionRight = (clientWidth - pageWidth) / 2 - adWidth - 10;
            var positionTop = document.documentElement.clientHeight / 2 - 20;

            if (G.browser.msie && parseInt(G.browser.version, 10) == 6) {
                two_dimensional.css({position: 'absolute',
                    top: (win_height/2 - two_dimensional.height() + doc_scrollTop - bottom) + 'px',
                    right : positionRight + 'px'
                });
            }
            else {
                two_dimensional.css({position: 'fixed' , top : positionTop +'px', right : positionRight + 'px'});
            }
        }

        jQuery('#feedback_link').unbind().bind('mouseover' , function(){
            jQuery(this).addClass('feedback_link_on');
        }).bind('mouseout' , function(){
            jQuery(this).removeClass('feedback_link_on');
        });
        jQuery('#two_dimensional_ul,#two_dimensional_feedback').hover(function(){
            jQuery(this).addClass('two_dimensional_hover');
        }, function(){
            jQuery(this).removeClass('two_dimensional_hover');
        });
        jQuery('#two_dimensional_x').click(function () {
            jQuery('#feedback_link').show();
            jQuery('#two_dimensional').hide();
            jQuery.cookie(C.COOKIE_TWODIMENSIONAL, '1', { path: '/', expires: 0.5});
        });

        feedback_position();
        two_dimensional_position();

        // 如果关闭了二维码
        if( jQuery.cookie(C.COOKIE_TWODIMENSIONAL) ){
            jQuery('#feedback_link').show();
        }else{
            jQuery('#two_dimensional').show();
        }

        jQuery(window).bind('scroll' , function(){
            feedback_position();
            two_dimensional_position();
        });
        jQuery(window).bind('resize' , function(){
            feedback_position();
            two_dimensional_position();
        });
    })();
});
</script>
